model{
      for(i in 2 : N){
         z[i] ~ dstable(alpha, beta, gamma, delta)   
         z[i] <- price[i] / price[i - 1] - 1
      }
      
      alpha ~ dunif(1.1, 2)
      beta ~ dunif(-1, 1)
      gamma ~ dunif(-0.05, 0.05)
      delta ~ dunif(0.001, 0.5)
      
      mean.z <- mean(z[2:50])
      sd.z <- sd(z[2:50])
   }
